Looking at your notebook, you already have the lemmatized corpus loaded as df with a tokens column and volume grouping. Here's a cell that generates the full exploratory wordlist:

python
#### Wordlist: All Unique Terms (Exploratory)
# Generates a full wordlist across the entire corpus with raw frequency, 
# relative frequency per 1,000,000 words, and TF-IDF score per volume.

from sklearn.feature_extraction.text import TfidfVectorizer
from collections import Counter

# --- 1. Corpus-wide raw + relative frequency ---
all_tokens = [t for tlist in df["tokens"] for t in tlist]
total_words = len(all_tokens)
raw_counts = Counter(all_tokens)

wordlist = pd.DataFrame([
    {
        "word": word,
        "raw_frequency": count,
        "relative_frequency_per_1M": (count / total_words) * 1_000_000
    }
    for word, count in raw_counts.items()
]).sort_values("raw_frequency", ascending=False).reset_index(drop=True)

print(f"Total unique words: {len(wordlist)}")
print(f"Total corpus words: {total_words:,}")
print(wordlist.head(30))
python
# --- 2. TF-IDF per volume (to flag distinctive vs. generic terms) ---
# Reuses volume_texts already built above

vectorizer_wl = TfidfVectorizer()
tfidf_matrix_wl = vectorizer_wl.fit_transform(volume_texts.values)
feature_names_wl = vectorizer_wl.get_feature_names_out()

df_tfidf_wl = pd.DataFrame(
    tfidf_matrix_wl.toarray(),
    index=volume_texts.index,
    columns=feature_names_wl
)

# Average TF-IDF score across all volumes per word
mean_tfidf = df_tfidf_wl.mean(axis=0).rename("mean_tfidf_across_volumes")

# Merge into wordlist
wordlist = wordlist.merge(mean_tfidf.reset_index().rename(columns={"index": "word"}), 
                          on="word", how="left")

wordlist = wordlist.sort_values("raw_frequency", ascending=False).reset_index(drop=True)

print(wordlist.head(50).to_string())
